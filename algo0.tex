\SetKwInput{Input}{Input}
\SetKwInOut{Output}{Output}
\SetKwFunction{distance}{distance}
\SetKwFunction{ESBA}{Epsilon\_step\_based\_approximation}
% for some reason, defaults getting overridden in RevTeX
\SetInd{0.5em}{1em}
\SetNlSkip{1em}
\Input{$Q$: 分解ターゲット行列}
\Input{$\epsilon$: 所望の分解精度}
\Input{$Q_1, Q_2... , Q_k$: $k$個のビルディングブロックの集合。具体的には、\\
- $Q_i = \lbrace U_{i1},...,U_{in} \rbrace$\\
- $U_{ij} \in$「Near-Identity」ユニタリー行列}

\Output{$\lbrace (j_1, j_2, ..., j_k) \mid D(\prod\limits_{i=1}^k U_{ij_{i}}, Q) \rbrace \leq \epsilon$}

\BlankLine
$ResultSet = \lbrace \rbrace$\\
\ForEach{$(j_1, j_2, ..., j_k) \in \lbrace (1, 1,... , 1) \rightarrow  (n, n,... , n)  \rbrace $} {
\tcc{全ての組み合わせを検証}
\If {$D(\prod\limits_{i=1}^k U_{ij_{i}}, Q) \rbrace \leq \epsilon$} {
	$ResultSet$に${(j_1, j_2, ..., j_n)}$を追加する
}
}
\Return{$ResultSet$}\\
